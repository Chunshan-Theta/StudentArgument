    新增題目
        |-> 
            選擇user_id                              -> SELECT * FROM `user_list`
            autoid: topic_id                         -> INSERT INTO `topic_list` (`topic_id`, `user_id`) VALUES (NULL, user_id);
            topic_list(topic_list)新增資料 ->|
    新增問卷
        |-> autoid: assessment_id                    -> INSERT INTO `assessment_list` (`assessment_id`) VALUES (NULL);
            assessment_list(assessment_list)新增資料 ->|

    建立活動
        |-> 選擇assessment:assessment_id             -> SELECT * FROM `assessment_list` 
            選擇host:user_id                         -> SELECT * FROM `user_list`
            選擇topic:topic_id                       -> SELECT * FROM `topic_list`
            autoid: activity_id                      ->INSERT INTO `activity_list` (`avtivity_id`, `topic_id`, `user_id`, `assessment_id`) VALUES (NULL, topic_id, user_id, assessment_id);
            activity_list(activity_list)新增資料     ->|
    設定活動聊天室
        |-> 選擇activity:activity_id                 -> SELECT * FROM `activity_list`
            輸入所需要聊天室數量                      -> input:NumOfChatroom
            autoid: chatroom_id(*=NumOfChatroom)     -> INSERT INTO `chatroom_list` (`chatroom_id`, `avtivity_id`) VALUES (NULL, activity_id);
            chatroom_list新增資料                    ->|
    

    新增活動學生
        |-> 選擇student:user_id                      -> SELECT * FROM `user_list`
            選擇activity:activity_id                 -> SELECT * FROM `activity_list`
            autoid: tester_id                        -> INSERT INTO `tester_list` (`tester_id`, `user_id`, `avtivity_id`, `chatroom_id`, `group`) VALUES (NULL, user_id, avtivity_id, DEFAULT, DEFAULT)
            tester_list新增資料                       ->| 

    安排活動學生進聊天室
        |-> 選擇activity:activity_id -> SELECT * FROM `activity_list`
            選擇chatroom:chatroom_id -> SELECT * FROM `chatroom_list` WHERE `avtivity_id` = activity_id
            選擇student:tester_id    -> SELECT * FROM `tester_list` WHERE `avtivity_id` = activity_id AND `chatroom_id` = -1
            update: chatroom_id      -> UPDATE `tester_list` SET `chatroom_id` = chatroom_id WHERE `tester_list`.`tester_id` = tester_id;
            tester_list 更新資料->|

            輸入活動代號(ActivityId):activity_id-> input:ActivityId
            選擇student:tester_id -> SELECT * FROM `tester_list` WHERE `user_id` = StuId AND `avtivity_id` = ActivityId
            回應聊天室代號:chatroom_id ->|
